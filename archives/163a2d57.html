js-学习案例：栏目切换
无标签
JavaScript
发布日期:   2019-11-07
阅读次数:  

tags:

开始记录学习的经过，今天做了一个tab栏目切换的小功能，整了很久才做出来，最后还想实现背景切换，但是没能做出来，今天先这样吧

仿照爱词霸的搜索，用js实现切换

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>搜索框切换功能</title>
    <style>
        body {
            background-image: url("banner-1x.png");
        }
        * {
            margin: 0;
            padding: 0;
            box-sizing: border-box;
        }
        .search {
            margin: 100px auto;
            width: 680px;
            position: relative;
        }
        .hd span {
            display: inline-block;
            width: 88px;
            height: 36px;
            background: rgba(0,0,0,.3);
            font-size: 16px;
            font-weight: 700;
            cursor: pointer;
            text-align: center;
            line-height: 40px;
        }
        .hd span.current{
            background-color: #fff;
        }
        input {
            width: 600px;
            height: 56px;
            outline: none;
            padding-left: 20px;
            color: gray;
            border: none;
            display: none;
            font-size: 18px;
        }
        input.current {
            display: block;
        }
        .sea-click {
            width: 80px;
            height: 56px;
            text-align: center;
            line-height: 56px;
            position: absolute;
            right: 0;
            bottom:0;
            background-color: #383838;
            color: #fff;
            cursor: pointer;
        }
    </style>
</head>
<body>
<div class="search" id="search">
    <div class="hd">
        <span class="current">查词</span>
        <span>翻译</span>
    </div>
    <div class="bd">
        <input type="text" value="请输入你要查询的单词" id="search_btn"  class="current"/>
        <input type="text" value="请输入你要查询的短语或句子" id="search_btn1"  />
        <div class="sea-click">搜索</div>
    </div>
</div>

<script>
    //先获取最外层的div
    var search = document.getElementById("search");
    //然后获取里面的第一个div
    var hd = search.getElementsByTagName("div")[0];
    //获取里面的第二个div
    var bd = search.getElementsByTagName("div")[1];
    //获取第一个div中所有的span标签
    var spans = hd.getElementsByTagName("span");
    //获取下方搜索框中的input标签
    var inputs = bd.getElementsByTagName("input");

    //循环遍历所有的span标签，并注册点击事件
    for (var i = 0; i < spans.length; i++) {
        //首先将span中的索引值存起来
        spans[i].setAttribute("index",i);
        //为每一个span注册点击事件
        spans[i].onclick = function () {
            //清除所有的span的类样式
            for (var j = 0; j < spans.length; j++) {
                spans[j].removeAttribute("class");
            }

            //给当前的span对象设置类样式
            this.className="current";
            //获取当前span对象的索引值
            var num = this.getAttribute("index");
            //清除所有的input的类样式
            for (var k = 0; k < inputs.length; k++) {
                inputs[k].removeAttribute("class");
            }
            //当前span对应的input标签设置类样式
            inputs[num].className = "current";
        };
    }



    for (var i = 0; i < inputs.length; i++) {
        //注册获取焦点的点击事件
        inputs[i].onfocus = function () {
            //判断当前对象的value值
            if (this.value == "请输入你要查询的单词" || "请输入你要查询的短语或句子"){
                this.value = "";
                this.style.color="black";
            }
        };
        //鼠标离开后的事件
        inputs[i].onblur = function () {
            //判断id值，设置value值
            if (this.id == "search_btn"){
                this.value = "请输入你要查询的单词";
                this.style.color="gray";

            }else {
                this.value = "请输入你要查询的短语或句子";
                this.style.color="gray";
            }
        };
    }


</script>

</body>
</html>

哎，整了一晚上，一开始就被布局给拦住了，css很多竟然又忘了。。爱词霸中查词和翻译那块是梯形，我也找到了一些做梯形的方法，可是在设置样式的时候总是出现问题，最后也放弃了继续深究，，只用普通的矩形来替代，先这样吧，整了一晚上才怎出个这么个玩意儿。。

但是不管怎么说也算是大概整明白排他功能。刚开始内容比较基础，但是觉得也可能比较重要，就先记录下来吧。
